-- activity.donate definition

CREATE TABLE `donate` (
  `id` bigint NOT NULL AUTO_INCREMENT COMMENT 'id',
  `doneter_id` bigint DEFAULT NULL COMMENT '捐赠人姓名',
  `donate_time` datetime DEFAULT NULL COMMENT '捐赠时间',
  `donate_type` varchar(100) DEFAULT NULL COMMENT '捐赠类型',
  `donate_destination` varchar(100) DEFAULT NULL COMMENT '捐赠地址',
  `donater_phone` varchar(100) DEFAULT NULL COMMENT '捐赠人联系电话',
  `receive_destination` varchar(100) DEFAULT NULL COMMENT '接收人地址',
  `receiver_phone` varchar(100) DEFAULT NULL COMMENT '接收人电话',
  `receiver_name` varchar(100) DEFAULT NULL COMMENT '接收人姓名',
  PRIMARY KEY (`id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci;
